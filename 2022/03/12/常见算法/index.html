冒泡排序

function sort(arr) {
  for (let i = 0; i < arr.length; i++) {
    for (let j = 0; j < arr.length - i; j++) {
      if (arr[j] > arr[j + 1]) {
        let t = arr[j]
        arr[j] = arr[j + 1]
        arr[j + 1] = t
      }
    }
  }
  return arr
}

输出当前日期

function getDate() {
  let today = new Date()

  let year = today.getFullYear()
  let month = today.getMonth() + 1
  let day = today.getDate()

  month < 10 ? '0' + month : month
  day < 10 ? '0' + day : day

  let date = year + '-' + month + '-' + day

  return date
}

字符串计数

let str = 'asdfssaaasasasasaa'
let json = {}

for (let i = 0; i < str.length; i++) {
  if (!json[str.charAt(i)]) {
    json[str.charAt(i)] = 1
  } else {
    json[str.charAt(i)]++
  }
}

let iMax = 0
let iIndex = ''

for (let i in json) {
  if (json[str.charAt(i)] > iMax) {
    iMax = json[i]
    iIndex = i
  }
}

console.log(iIndex + '-' + iMax);
